```typescript
import { api } from '../api';

// Types
export interface ImplementationProject {
  id: string;
  organizationId: string;
  organizationName: string;
  type: 'QUICK_START' | 'STANDARD' | 'COMPLEX' | 'ENTERPRISE';
  status: string;
  name: string;
  specialistId: string;
  specialistName: string;
  managerId?: string;
  managerName?: string;
  goLiveDate?: string;
  healthScore: number;
  totalTasks: number;
  completedTasks: number;
  blockedTasks: number;
  createdAt: string;
  updatedAt: string;
}

export interface ImplementationTask {
  id: string;
  projectId: string;
  phase: string;
  name: string;
  description?: string;
  status: 'PENDING' | 'IN_PROGRESS' | 'COMPLETED' | 'BLOCKED' | 'SKIPPED';
  isRequired: boolean;
  sortOrder: number;
  dueDate?: string;
  completedAt?: string;
  completedByName?: string;
  notes?: string;
}

export interface ImplementationBlocker {
  id: string;
  projectId: string;
  taskId?: string;
  taskName?: string;
  title: string;
  description: string;
  severity: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
  isResolved: boolean;
  resolvedAt?: string;
  resolution?: string;
  createdByName: string;
  createdAt: string;
}

export interface ReadinessReport {
  projectId: string;
  projectName: string;
  overallScore: number;
  isReady: boolean;
  categories: {
    category: string;
    name: string;
    isReady: boolean;
    score: number;
    issues: string[];
  }[];
  blockingIssues: string[];
  signOffs: {
    type: string;
    signedByName: string;
    signedAt: string;
  }[];
}

export const implementationApi = {
  // Projects
  listProjects: (params?: { status?: string; specialistId?: string; type?: string }) =>
    api.get<{ projects: ImplementationProject[]; total: number }>('/internal/implementation/projects', { params }),

  getProject: (id: string) =>
    api.get<ImplementationProject & { tasks: ImplementationTask[]; blockers: ImplementationBlocker[] }>(
      `/internal/implementation/projects/${id}`
    ),

  createProject: (data: { organizationId: string; type: string; name: string; specialistId: string; goLiveDate?: string }) =>
    api.post<ImplementationProject>('/internal/implementation/projects', data),

  updateProject: (id: string, data: Partial<ImplementationProject>) =>
    api.patch<ImplementationProject>(`/internal/implementation/projects/${id}`, data),

  // Tasks
  updateTaskStatus: (taskId: string, status: string, notes?: string) =>
    api.patch(`/internal/implementation/tasks/${taskId}/status`, { status, notes }),

  // Blockers
  createBlocker: (data: { projectId: string; taskId?: string; title: string; description: string; severity: string }) =>
    api.post<ImplementationBlocker>('/internal/implementation/blockers', data),

  resolveBlocker: (blockerId: string, resolution: string) =>
    api.patch(`/internal/implementation/blockers/${blockerId}/resolve`, { resolution }),

  // Go-Live Readiness
  getReadiness: (projectId: string) =>
    api.get<ReadinessReport>(`/internal/implementation/projects/${projectId}/readiness`),

  recordSignOff: (projectId: string, type: string, notes?: string) =>
    api.post(`/internal/implementation/projects/${projectId}/sign-offs`, { type, notes }),

  getSignOffs: (projectId: string) =>
    api.get(`/internal/implementation/projects/${projectId}/sign-offs`),
};
```

```typescript
import { useState } from 'react';
import { useQuery } from '@tanstack/react-query';
import { useNavigate } from 'react-router-dom';
import { implementationApi, ImplementationProject } from '@/services/internal/implementation.api';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';
import { Badge } from '@/components/ui/badge';
import { Progress } from '@/components/ui/progress';
import { Plus, Calendar, AlertTriangle, CheckCircle2 } from 'lucide-react';

export function ImplementationDashboard() {
  const navigate = useNavigate();
  const [statusFilter, setStatusFilter] = useState<string>('');

  const { data, isLoading } = useQuery({
    queryKey: ['implementation-projects', statusFilter],
    queryFn: () => implementationApi.listProjects({ status: statusFilter || undefined }),
  });

  const getHealthColor = (score: number) => {
    if (score >= 80) return 'text-green-600';
    if (score >= 60) return 'text-yellow-600';
    return 'text-red-600';
  };

  const getStatusBadge = (status: string) => {
    const variants: Record<string, 'default' | 'secondary' | 'destructive' | 'outline'> = {
      NOT_STARTED: 'outline',
      DISCOVERY: 'secondary',
      CONFIGURATION: 'secondary',
      TRAINING: 'secondary',
      GO_LIVE_PREP: 'default',
      COMPLETED: 'default',
      ON_HOLD: 'destructive',
    };
    return <Badge variant={variants[status] || 'outline'}>{status.replace('_', ' ')}</Badge>;
  };

  return (
    <div className="p-6 space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">Implementation Projects</h1>
          <p className="text-gray-500">Manage client onboarding and go-live readiness</p>
        </div>
        <Button onClick={() => navigate('/internal/implementation/new')}>
          <Plus className="h-4 w-4 mr-2" />
          New Project
        </Button>
      </div>

      {/* Filters */}
      <div className="flex items-center gap-4">
        <Select value={statusFilter} onValueChange={setStatusFilter}>
          <SelectTrigger className="w-48">
            <SelectValue placeholder="All Statuses" />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="">All Statuses</SelectItem>
            <SelectItem value="NOT_STARTED">Not Started</SelectItem>
            <SelectItem value="DISCOVERY">Discovery</SelectItem>
            <SelectItem value="CONFIGURATION">Configuration</SelectItem>
            <SelectItem value="TRAINING">Training</SelectItem>
            <SelectItem value="GO_LIVE_PREP">Go-Live Prep</SelectItem>
            <SelectItem value="COMPLETED">Completed</SelectItem>
            <SelectItem value="ON_HOLD">On Hold</SelectItem>
          </SelectContent>
        </Select>
      </div>

      {/* Project Grid */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {data?.projects.map((project) => (
          <Card
            key={project.id}
            className="cursor-pointer hover:shadow-md transition-shadow"
            onClick={() => navigate(`/internal/implementation/${project.id}`)}
          >
            <CardHeader className="pb-2">
              <div className="flex items-start justify-between">
                <div>
                  <CardTitle className="text-lg">{project.name}</CardTitle>
                  <p className="text-sm text-gray-500">{project.organizationName}</p>
                </div>
                {getStatusBadge(project.status)}
              </div>
            </CardHeader>
            <CardContent className="space-y-4">
              {/* Progress */}
              <div className="space-y-1">
                <div className="flex justify-between text-sm">
                  <span>Progress</span>
                  <span>{Math.round((project.completedTasks / project.totalTasks) * 100)}%</span>
                </div>
                <Progress value={(project.completedTasks / project.totalTasks) * 100} />
              </div>

              {/* Stats */}
              <div className="flex justify-between text-sm">
                <div className="flex items-center gap-1">
                  <CheckCircle2 className="h-4 w-4 text-green-600" />
                  {project.completedTasks}/{project.totalTasks} tasks
                </div>
                {project.blockedTasks > 0 && (
                  <div className="flex items-center gap-1 text-red-600">
                    <AlertTriangle className="h-4 w-4" />
                    {project.blockedTasks} blocked
                  </div>
                )}
                <div className={getHealthColor(project.healthScore)}>
                  {project.healthScore}% health
                </div>
              </div>

              {/* Go-Live Date */}
              {project.goLiveDate && (
                <div className="flex items-center gap-2 text-sm text-gray-500">
                  <Calendar className="h-4 w-4" />
                  Go-live: {new Date(project.goLiveDate).toLocaleDateString()}
                </div>
              )}
            </CardContent>
          </Card>
        ))}
      </div>
    </div>
  );
}
```

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { implementationApi, ReadinessReport } from '@/services/internal/implementation.api';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Progress } from '@/components/ui/progress';
import { Badge } from '@/components/ui/badge';
import { CheckCircle2, XCircle, AlertTriangle, FileSignature } from 'lucide-react';
import { cn } from '@/lib/utils';

interface GoLiveReadinessProps {
  projectId: string;
}

export function GoLiveReadiness({ projectId }: GoLiveReadinessProps) {
  const queryClient = useQueryClient();

  const { data: readiness, isLoading } = useQuery({
    queryKey: ['readiness', projectId],
    queryFn: () => implementationApi.getReadiness(projectId),
  });

  const signOffMutation = useMutation({
    mutationFn: ({ type, notes }: { type: string; notes?: string }) =>
      implementationApi.recordSignOff(projectId, type, notes),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['readiness', projectId] });
    },
  });

  if (isLoading) return <div>Loading...</div>;
  if (!readiness) return null;

  const getCategoryIcon = (isReady: boolean) =>
    isReady ? (
      <CheckCircle2 className="h-5 w-5 text-green-600" />
    ) : (
      <XCircle className="h-5 w-5 text-red-600" />
    );

  return (
    <div className="space-y-6">
      {/* Overall Score */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center justify-between">
            <span>Go-Live Readiness</span>
            <Badge
              variant={readiness.isReady ? 'default' : 'destructive'}
              className="text-lg px-4 py-1"
            >
              {readiness.isReady ? 'READY' : 'NOT READY'}
            </Badge>
          </CardTitle>
        </CardHeader>
        <CardContent>
          <div className="space-y-2">
            <div className="flex justify-between">
              <span className="text-2xl font-bold">{readiness.overallScore}%</span>
              <span className="text-gray-500">Overall Score</span>
            </div>
            <Progress
              value={readiness.overallScore}
              className={cn(
                readiness.overallScore >= 80 ? 'bg-green-100' :
                readiness.overallScore >= 60 ? 'bg-yellow-100' : 'bg-red-100'
              )}
            />
          </div>
        </CardContent>
      </Card>

      {/* Category Breakdown */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {readiness.categories.map((cat) => (
          <Card key={cat.category}>
            <CardHeader className="pb-2">
              <CardTitle className="text-sm flex items-center gap-2">
                {getCategoryIcon(cat.isReady)}
                {cat.name}
              </CardTitle>
            </CardHeader>
            <CardContent>
              <div className="space-y-2">
                <Progress value={cat.score} className="h-2" />
                <div className="text-sm text-gray-500">{cat.score}% complete</div>
                {cat.issues.length > 0 && (
                  <div className="space-y-1">
                    {cat.issues.slice(0, 3).map((issue, i) => (
                      <div key={i} className="flex items-center gap-1 text-xs text-red-600">
                        <AlertTriangle className="h-3 w-3" />
                        {issue}
                      </div>
                    ))}
                    {cat.issues.length > 3 && (
                      <div className="text-xs text-gray-500">
                        +{cat.issues.length - 3} more issues
                      </div>
                    )}
                  </div>
                )}
              </div>
            </CardContent>
          </Card>
        ))}
      </div>

      {/* Sign-offs */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <FileSignature className="h-5 w-5" />
            Required Sign-offs
          </CardTitle>
        </CardHeader>
        <CardContent>
          <div className="space-y-4">
            {['CLIENT_ADMIN', 'IMPLEMENTATION_SPECIALIST'].map((type) => {
              const signOff = readiness.signOffs.find((s) => s.type === type);
              return (
                <div key={type} className="flex items-center justify-between">
                  <div className="flex items-center gap-2">
                    {signOff ? (
                      <CheckCircle2 className="h-5 w-5 text-green-600" />
                    ) : (
                      <div className="h-5 w-5 rounded-full border-2 border-gray-300" />
                    )}
                    <span>{type.replace('_', ' ')}</span>
                  </div>
                  {signOff ? (
                    <span className="text-sm text-gray-500">
                      {signOff.signedByName} - {new Date(signOff.signedAt).toLocaleDateString()}
                    </span>
                  ) : (
                    <Button
                      size="sm"
                      variant="outline"
                      onClick={() => signOffMutation.mutate({ type })}
                    >
                      Record Sign-off
                    </Button>
                  )}
                </div>
              );
            })}
          </div>
        </CardContent>
      </Card>
    </div>
  );
}
```
