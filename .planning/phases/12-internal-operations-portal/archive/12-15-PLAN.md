import { api } from '../api';

export type HealthTrend = 'IMPROVING' | 'STABLE' | 'DECLINING';
export type RiskLevel = 'LOW' | 'MEDIUM' | 'HIGH';

export interface TenantHealthScore {
  id: string;
  organizationId: string;
  organizationName: string;
  calculatedAt: string;
  overallScore: number;
  trend: HealthTrend;
  riskLevel: RiskLevel;
  components: {
    loginScore: number;
    caseResolutionScore: number;
    campaignCompletionScore: number;
    featureAdoptionScore: number;
    supportTicketScore: number;
  };
  renewalDate?: string;
  contractValue?: number;
}

export interface HealthScoreHistory {
  date: string;
  overallScore: number;
  trend: HealthTrend;
  riskLevel: RiskLevel;
}

export interface UsageMetric {
  date: string;
  value: number;
  changePercent?: number;
}

export interface FeatureAdoption {
  featureName: string;
  category: string;
  isEnabled: boolean;
  isUsed: boolean;
  lastUsedAt?: string;
  usageCount?: number;
}

export interface RenewalRiskSummary {
  totalTenants: number;
  highRisk: number;
  mediumRisk: number;
  lowRisk: number;
  tenantsAtRisk: {
    organizationId: string;
    organizationName: string;
    riskLevel: RiskLevel;
    healthScore: number;
    renewalDate?: string;
    contractValue?: number;
    primaryIssues: string[];
  }[];
}

export const clientSuccessApi = {
  // Health Scores
  listHealthScores: (params?: { riskLevel?: RiskLevel; minScore?: number; maxScore?: number; page?: number; limit?: number }) =>
    api.get<{ items: TenantHealthScore[]; total: number }>('/internal/client-health/scores', { params }),

  getHealthScore: (organizationId: string) =>
    api.get<TenantHealthScore>(`/internal/client-health/scores/${organizationId}`),

  getHealthHistory: (organizationId: string, days: number = 30) =>
    api.get<HealthScoreHistory[]>(`/internal/client-health/scores/${organizationId}/history`, { params: { days } }),

  recalculateScore: (organizationId: string) =>
    api.post(`/internal/client-health/scores/${organizationId}/recalculate`),

  // Usage Metrics
  getLoginMetrics: (organizationId: string, days: number = 30) =>
    api.get<UsageMetric[]>(`/internal/client-health/metrics/${organizationId}`, {
      params: { metricType: 'LOGIN', days },
    }),

  getCaseMetrics: (organizationId: string, days: number = 30) =>
    api.get<UsageMetric[]>(`/internal/client-health/metrics/${organizationId}`, {
      params: { metricType: 'CASE_RESOLUTION', days },
    }),

  getCampaignMetrics: (organizationId: string, days: number = 30) =>
    api.get<UsageMetric[]>(`/internal/client-health/metrics/${organizationId}`, {
      params: { metricType: 'CAMPAIGN_COMPLETION', days },
    }),

  // Feature Adoption
  getFeatureAdoption: (organizationId: string) =>
    api.get<FeatureAdoption[]>(`/internal/client-health/adoption/${organizationId}`),

  // Renewal Risk
  getRenewalRiskSummary: () =>
    api.get<RenewalRiskSummary>('/internal/client-health/renewal-risk'),

  // Bulk operations
  calculateAllScores: () =>
    api.post('/internal/client-health/calculate-all'),
};

import { useQuery } from '@tanstack/react-query';
import { clientSuccessApi, TenantHealthScore, HealthTrend, RiskLevel } from '@/services/internal/client-success.api';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Progress } from '@/components/ui/progress';
import { TrendingUp, TrendingDown, Minus, AlertTriangle, CheckCircle2 } from 'lucide-react';
import { cn } from '@/lib/utils';

interface TenantHealthCardProps {
  health: TenantHealthScore;
  onClick?: () => void;
  showDetails?: boolean;
}

const HEALTH_WEIGHTS = {
  login: 0.20,
  caseResolution: 0.25,
  campaignCompletion: 0.25,
  featureAdoption: 0.15,
  supportTickets: 0.15,
};

const componentLabels: Record<string, string> = {
  loginScore: 'User Logins',
  caseResolutionScore: 'Case Resolution',
  campaignCompletionScore: 'Campaign Completion',
  featureAdoptionScore: 'Feature Adoption',
  supportTicketScore: 'Support Tickets',
};

export function TenantHealthCard({ health, onClick, showDetails = false }: TenantHealthCardProps) {
  const getTrendIcon = (trend: HealthTrend) => {
    switch (trend) {
      case 'IMPROVING':
        return <TrendingUp className="h-4 w-4 text-green-600" />;
      case 'DECLINING':
        return <TrendingDown className="h-4 w-4 text-red-600" />;
      default:
        return <Minus className="h-4 w-4 text-gray-400" />;
    }
  };

  const getRiskBadge = (risk: RiskLevel) => {
    const variants: Record<RiskLevel, 'default' | 'secondary' | 'destructive'> = {
      LOW: 'default',
      MEDIUM: 'secondary',
      HIGH: 'destructive',
    };
    return <Badge variant={variants[risk]}>{risk} Risk</Badge>;
  };

  const getScoreColor = (score: number) => {
    if (score >= 80) return 'bg-green-500';
    if (score >= 60) return 'bg-yellow-500';
    return 'bg-red-500';
  };

  return (
    <Card
      className={cn("transition-shadow", onClick && "cursor-pointer hover:shadow-md")}
      onClick={onClick}
    >
      <CardHeader className="pb-2">
        <div className="flex items-start justify-between">
          <div>
            <CardTitle className="text-lg">{health.organizationName}</CardTitle>
            <p className="text-sm text-gray-500">
              Last updated: {new Date(health.calculatedAt).toLocaleDateString()}
            </p>
          </div>
          <div className="flex items-center gap-2">
            {getTrendIcon(health.trend)}
            {getRiskBadge(health.riskLevel)}
          </div>
        </div>
      </CardHeader>
      <CardContent className="space-y-4">
        {/* Overall Score */}
        <div className="flex items-center gap-4">
          <div className={cn(
            "w-16 h-16 rounded-full flex items-center justify-center text-white font-bold text-xl",
            getScoreColor(health.overallScore)
          )}>
            {health.overallScore}
          </div>
          <div className="flex-1">
            <Progress value={health.overallScore} className="h-3" />
            <p className="text-sm text-gray-500 mt-1">Overall Health Score</p>
          </div>
        </div>

        {/* Component Breakdown */}
        {showDetails && (
          <div className="space-y-3 pt-4 border-t">
            <h4 className="font-medium text-sm">Component Scores</h4>
            {Object.entries(health.components).map(([key, value]) => (
              <div key={key} className="space-y-1">
                <div className="flex justify-between text-sm">
                  <span>{componentLabels[key]}</span>
                  <span className={cn(
                    "font-medium",
                    value >= 80 ? "text-green-600" :
                    value >= 60 ? "text-yellow-600" : "text-red-600"
                  )}>
                    {value}%
                  </span>
                </div>
                <Progress value={value} className="h-2" />
              </div>
            ))}
          </div>
        )}

        {/* Renewal Info */}
        {health.renewalDate && (
          <div className="flex items-center justify-between text-sm pt-2 border-t">
            <span className="text-gray-500">Renewal Date</span>
            <span className="font-medium">{new Date(health.renewalDate).toLocaleDateString()}</span>
          </div>
        )}
      </CardContent>
    </Card>
  );
}

import { useState } from 'react';
import { useQuery } from '@tanstack/react-query';
import { clientSuccessApi, UsageMetric } from '@/services/internal/client-success.api';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  ResponsiveContainer,
  AreaChart,
  Area,
} from 'recharts';

interface UsageChartsProps {
  organizationId: string;
}

export function UsageCharts({ organizationId }: UsageChartsProps) {
  const [days, setDays] = useState(30);
  const [activeTab, setActiveTab] = useState('logins');

  const { data: loginMetrics } = useQuery({
    queryKey: ['usage-metrics', organizationId, 'login', days],
    queryFn: () => clientSuccessApi.getLoginMetrics(organizationId, days),
  });

  const { data: caseMetrics } = useQuery({
    queryKey: ['usage-metrics', organizationId, 'case', days],
    queryFn: () => clientSuccessApi.getCaseMetrics(organizationId, days),
  });

  const { data: campaignMetrics } = useQuery({
    queryKey: ['usage-metrics', organizationId, 'campaign', days],
    queryFn: () => clientSuccessApi.getCampaignMetrics(organizationId, days),
  });

  const formatDate = (dateStr: string) => {
    const date = new Date(dateStr);
    return `${date.getMonth() + 1}/${date.getDate()}`;
  };

  const renderChart = (data: UsageMetric[] | undefined, color: string, label: string) => {
    if (!data || data.length === 0) {
      return <div className="h-64 flex items-center justify-center text-gray-500">No data available</div>;
    }

    return (
      <ResponsiveContainer width="100%" height={256}>
        <AreaChart data={data}>
          <CartesianGrid strokeDasharray="3 3" />
          <XAxis dataKey="date" tickFormatter={formatDate} />
          <YAxis />
          <Tooltip
            labelFormatter={(value) => new Date(value).toLocaleDateString()}
            formatter={(value: number) => [`${value}%`, label]}
          />
          <Area
            type="monotone"
            dataKey="value"
            stroke={color}
            fill={color}
            fillOpacity={0.2}
          />
        </AreaChart>
      </ResponsiveContainer>
    );
  };

  return (
    <Card>
      <CardHeader>
        <div className="flex items-center justify-between">
          <CardTitle>Usage Trends</CardTitle>
          <Select value={days.toString()} onValueChange={(v) => setDays(parseInt(v))}>
            <SelectTrigger className="w-32">
              <SelectValue />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="7">7 days</SelectItem>
              <SelectItem value="30">30 days</SelectItem>
              <SelectItem value="90">90 days</SelectItem>
            </SelectContent>
          </Select>
        </div>
      </CardHeader>
      <CardContent>
        <Tabs value={activeTab} onValueChange={setActiveTab}>
          <TabsList>
            <TabsTrigger value="logins">User Logins</TabsTrigger>
            <TabsTrigger value="cases">Case Resolution</TabsTrigger>
            <TabsTrigger value="campaigns">Campaigns</TabsTrigger>
          </TabsList>

          <TabsContent value="logins" className="mt-4">
            {renderChart(loginMetrics, '#3b82f6', 'Login Rate')}
          </TabsContent>

          <TabsContent value="cases" className="mt-4">
            {renderChart(caseMetrics, '#22c55e', 'On-Time Rate')}
          </TabsContent>

          <TabsContent value="campaigns" className="mt-4">
            {renderChart(campaignMetrics, '#8b5cf6', 'Completion Rate')}
          </TabsContent>
        </Tabs>
      </CardContent>
    </Card>
  );
}
