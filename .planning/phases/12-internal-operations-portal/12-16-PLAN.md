```tsx
'use client';

import { useState } from 'react';
import { useQuery } from '@tanstack/react-query';
import Link from 'next/link';
import { HealthScoreCard } from '@/components/client-success/HealthScoreCard';
import { TrendingUp, TrendingDown, Minus, AlertTriangle, Building2, Filter } from 'lucide-react';
import { cn } from '@/lib/utils';

type HealthFilter = 'all' | 'red' | 'amber' | 'green';

export default function ClientSuccessDashboardPage() {
  const [healthFilter, setHealthFilter] = useState<HealthFilter>('all');
  const [sortBy, setSortBy] = useState<'score' | 'name' | 'trend'>('score');

  const { data, isLoading } = useQuery({
    queryKey: ['client-health-portfolio'],
    queryFn: async () => {
      const res = await fetch('/api/v1/internal/client-success/portfolio');
      return res.json();
    },
  });

  const getHealthColor = (score: number): 'red' | 'amber' | 'green' => {
    if (score >= 80) return 'green';
    if (score >= 60) return 'amber';
    return 'red';
  };

  const filteredClients = data?.clients?.filter((client: any) => {
    if (healthFilter === 'all') return true;
    return getHealthColor(client.healthScore) === healthFilter;
  }).sort((a: any, b: any) => {
    if (sortBy === 'score') return b.healthScore - a.healthScore;
    if (sortBy === 'name') return a.name.localeCompare(b.name);
    if (sortBy === 'trend') return b.trend - a.trend;
    return 0;
  });

  const healthCounts = {
    red: data?.clients?.filter((c: any) => c.healthScore < 60).length || 0,
    amber: data?.clients?.filter((c: any) => c.healthScore >= 60 && c.healthScore < 80).length || 0,
    green: data?.clients?.filter((c: any) => c.healthScore >= 80).length || 0,
  };

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">Client Success Dashboard</h1>
          <p className="text-gray-500">Portfolio health and usage metrics</p>
        </div>
        <Link
          href="/client-success/benchmarks"
          className="px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700"
        >
          View Benchmarks
        </Link>
      </div>

      {/* Portfolio Summary - Traffic Light View */}
      <div className="grid grid-cols-4 gap-4">
        <button
          onClick={() => setHealthFilter('all')}
          className={cn(
            'p-4 bg-white border rounded-lg text-left',
            healthFilter === 'all' && 'ring-2 ring-blue-500'
          )}
        >
          <div className="text-2xl font-bold">{data?.clients?.length || 0}</div>
          <div className="text-sm text-gray-500">Total Clients</div>
        </button>
        <button
          onClick={() => setHealthFilter('green')}
          className={cn(
            'p-4 bg-green-50 border border-green-200 rounded-lg text-left',
            healthFilter === 'green' && 'ring-2 ring-green-500'
          )}
        >
          <div className="text-2xl font-bold text-green-700">{healthCounts.green}</div>
          <div className="text-sm text-green-600">Healthy (80%+)</div>
        </button>
        <button
          onClick={() => setHealthFilter('amber')}
          className={cn(
            'p-4 bg-yellow-50 border border-yellow-200 rounded-lg text-left',
            healthFilter === 'amber' && 'ring-2 ring-yellow-500'
          )}
        >
          <div className="text-2xl font-bold text-yellow-700">{healthCounts.amber}</div>
          <div className="text-sm text-yellow-600">At Risk (60-79%)</div>
        </button>
        <button
          onClick={() => setHealthFilter('red')}
          className={cn(
            'p-4 bg-red-50 border border-red-200 rounded-lg text-left',
            healthFilter === 'red' && 'ring-2 ring-red-500'
          )}
        >
          <div className="text-2xl font-bold text-red-700">{healthCounts.red}</div>
          <div className="text-sm text-red-600">Critical (&lt;60%)</div>
        </button>
      </div>

      {/* Sort controls */}
      <div className="flex items-center gap-2">
        <Filter className="h-4 w-4 text-gray-500" />
        <span className="text-sm text-gray-500">Sort by:</span>
        {(['score', 'name', 'trend'] as const).map((option) => (
          <button
            key={option}
            onClick={() => setSortBy(option)}
            className={cn(
              'px-3 py-1 text-sm rounded-full',
              sortBy === option
                ? 'bg-blue-100 text-blue-700'
                : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
            )}
          >
            {option.charAt(0).toUpperCase() + option.slice(1)}
          </button>
        ))}
      </div>

      {/* Client List */}
      {isLoading ? (
        <div className="text-center py-8">Loading...</div>
      ) : (
        <div className="grid grid-cols-2 gap-4">
          {filteredClients?.map((client: any) => (
            <HealthScoreCard key={client.id} client={client} />
          ))}
        </div>
      )}
    </div>
  );
}
```

```tsx
'use client';

import { useState } from 'react';
import Link from 'next/link';
import { Building2, TrendingUp, TrendingDown, Minus, ChevronDown, ChevronUp, Bell, BellOff } from 'lucide-react';
import { cn } from '@/lib/utils';

interface HealthScoreCardProps {
  client: {
    id: string;
    name: string;
    healthScore: number;
    trend: number; // -100 to +100
    components: {
      login: number;        // 20% weight
      caseResolution: number; // 25% weight
      campaignCompletion: number; // 25% weight
      featureAdoption: number; // 15% weight
      ticketVolume: number; // 15% weight (inverse - lower is better)
    };
    alertsEnabled: boolean;
    lastActivity: string;
  };
}

export function HealthScoreCard({ client }: HealthScoreCardProps) {
  const [expanded, setExpanded] = useState(false);

  const getHealthColor = (score: number) => {
    if (score >= 80) return { bg: 'bg-green-500', text: 'text-green-700', light: 'bg-green-50' };
    if (score >= 60) return { bg: 'bg-yellow-500', text: 'text-yellow-700', light: 'bg-yellow-50' };
    return { bg: 'bg-red-500', text: 'text-red-700', light: 'bg-red-50' };
  };

  const colors = getHealthColor(client.healthScore);
  const TrendIcon = client.trend > 0 ? TrendingUp : client.trend < 0 ? TrendingDown : Minus;
  const trendColor = client.trend > 0 ? 'text-green-600' : client.trend < 0 ? 'text-red-600' : 'text-gray-500';

  const componentWeights = [
    { key: 'login', label: 'Login Activity', weight: '20%', value: client.components.login },
    { key: 'caseResolution', label: 'Case Resolution', weight: '25%', value: client.components.caseResolution },
    { key: 'campaignCompletion', label: 'Campaign Completion', weight: '25%', value: client.components.campaignCompletion },
    { key: 'featureAdoption', label: 'Feature Adoption', weight: '15%', value: client.components.featureAdoption },
    { key: 'ticketVolume', label: 'Support Tickets (inv)', weight: '15%', value: client.components.ticketVolume },
  ];

  return (
    <div className={cn('border rounded-lg overflow-hidden', colors.light)}>
      {/* Main card */}
      <div className="p-4">
        <div className="flex items-center justify-between">
          <Link
            href={`/client-success/${client.id}`}
            className="flex items-center gap-3 hover:underline"
          >
            <Building2 className="h-5 w-5 text-gray-400" />
            <div>
              <div className="font-medium">{client.name}</div>
              <div className="text-xs text-gray-500">
                Last activity: {new Date(client.lastActivity).toLocaleDateString()}
              </div>
            </div>
          </Link>

          {/* Health Score with Traffic Light */}
          <div className="flex items-center gap-4">
            {/* Trend */}
            <div className={cn('flex items-center gap-1', trendColor)}>
              <TrendIcon className="h-4 w-4" />
              <span className="text-sm font-medium">{Math.abs(client.trend)}%</span>
            </div>

            {/* Alert status */}
            {client.alertsEnabled ? (
              <Bell className="h-4 w-4 text-blue-500" />
            ) : (
              <BellOff className="h-4 w-4 text-gray-300" />
            )}

            {/* Score circle */}
            <div
              className={cn(
                'w-14 h-14 rounded-full flex items-center justify-center text-white font-bold',
                colors.bg
              )}
            >
              {client.healthScore}
            </div>
          </div>
        </div>

        {/* Expand button */}
        <button
          onClick={() => setExpanded(!expanded)}
          className="mt-3 flex items-center gap-1 text-sm text-gray-500 hover:text-gray-700"
        >
          {expanded ? <ChevronUp className="h-4 w-4" /> : <ChevronDown className="h-4 w-4" />}
          {expanded ? 'Hide details' : 'Show score breakdown'}
        </button>
      </div>

      {/* Expanded detail - drill-down per CONTEXT.md */}
      {expanded && (
        <div className="border-t bg-white p-4 space-y-3">
          {componentWeights.map((comp) => {
            const compColors = getHealthColor(comp.value);
            return (
              <div key={comp.key} className="flex items-center justify-between">
                <div className="text-sm">
                  <span className="font-medium">{comp.label}</span>
                  <span className="text-gray-400 ml-2">({comp.weight})</span>
                </div>
                <div className="flex items-center gap-2">
                  <div className="w-24 h-2 bg-gray-200 rounded-full overflow-hidden">
                    <div
                      className={cn('h-full', compColors.bg)}
                      style={{ width: `${comp.value}%` }}
                    />
                  </div>
                  <span className={cn('text-sm font-medium w-10 text-right', compColors.text)}>
                    {comp.value}%
                  </span>
                </div>
              </div>
            );
          })}
        </div>
      )}
    </div>
  );
}
```

```tsx
'use client';

import { useParams } from 'next/navigation';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import Link from 'next/link';
import { UsageMetricsChart } from '@/components/client-success/UsageMetricsChart';
import { FeatureAdoptionGrid } from '@/components/client-success/FeatureAdoptionGrid';
import { ArrowLeft, Bell, BellOff, Settings, Calendar } from 'lucide-react';
import { cn } from '@/lib/utils';
import { format } from 'date-fns';

export default function ClientDetailPage() {
  const { orgId } = useParams();
  const queryClient = useQueryClient();

  const { data: client, isLoading } = useQuery({
    queryKey: ['client-health', orgId],
    queryFn: async () => {
      const res = await fetch(`/api/v1/internal/client-success/health/${orgId}`);
      return res.json();
    },
  });

  const { data: usageData } = useQuery({
    queryKey: ['client-usage', orgId],
    queryFn: async () => {
      const res = await fetch(`/api/v1/internal/client-success/usage/${orgId}`);
      return res.json();
    },
  });

  const { data: features } = useQuery({
    queryKey: ['client-features', orgId],
    queryFn: async () => {
      const res = await fetch(`/api/v1/internal/client-success/features/${orgId}`);
      return res.json();
    },
  });

  const toggleAlertsMutation = useMutation({
    mutationFn: async (enabled: boolean) => {
      const res = await fetch(`/api/v1/internal/client-success/alerts/${orgId}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ enabled }),
      });
      return res.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['client-health', orgId] });
    },
  });

  if (isLoading) return <div>Loading...</div>;

  const healthColor = client?.healthScore >= 80
    ? 'bg-green-500'
    : client?.healthScore >= 60
    ? 'bg-yellow-500'
    : 'bg-red-500';

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center gap-4">
        <Link href="/client-success" className="p-2 hover:bg-gray-100 rounded">
          <ArrowLeft className="h-5 w-5" />
        </Link>
        <div className="flex-1">
          <h1 className="text-2xl font-bold">{client?.name}</h1>
          <p className="text-gray-500">Client health and usage details</p>
        </div>
        <div className={cn('w-16 h-16 rounded-full flex items-center justify-center text-white font-bold text-xl', healthColor)}>
          {client?.healthScore}
        </div>
      </div>

      {/* Alert Settings */}
      <div className="bg-white border rounded-lg p-4 flex items-center justify-between">
        <div className="flex items-center gap-3">
          {client?.alertsEnabled ? (
            <Bell className="h-5 w-5 text-blue-500" />
          ) : (
            <BellOff className="h-5 w-5 text-gray-400" />
          )}
          <div>
            <div className="font-medium">Proactive Alerts</div>
            <div className="text-sm text-gray-500">
              {client?.alertsEnabled
                ? 'You will be notified when health score drops'
                : 'Alerts disabled (PLG/SMB mode)'}
            </div>
          </div>
        </div>
        <button
          onClick={() => toggleAlertsMutation.mutate(!client?.alertsEnabled)}
          className={cn(
            'px-4 py-2 rounded-lg text-sm',
            client?.alertsEnabled
              ? 'bg-gray-100 text-gray-700 hover:bg-gray-200'
              : 'bg-blue-600 text-white hover:bg-blue-700'
          )}
        >
          {client?.alertsEnabled ? 'Disable Alerts' : 'Enable Alerts'}
        </button>
      </div>

      {/* Usage Metrics Chart */}
      <div className="bg-white border rounded-lg p-6">
        <h2 className="text-lg font-semibold mb-4">Usage Metrics (30 days)</h2>
        <UsageMetricsChart data={usageData} />
      </div>

      {/* Feature Adoption */}
      <div className="bg-white border rounded-lg p-6">
        <h2 className="text-lg font-semibold mb-4">Feature Adoption</h2>
        <FeatureAdoptionGrid features={features} />
      </div>

      {/* Recent Activity */}
      <div className="bg-white border rounded-lg p-6">
        <h2 className="text-lg font-semibold mb-4">Recent Activity</h2>
        <div className="space-y-3">
          {client?.recentActivity?.map((activity: any, i: number) => (
            <div key={i} className="flex items-center gap-3 text-sm">
              <Calendar className="h-4 w-4 text-gray-400" />
              <span className="text-gray-500">
                {format(new Date(activity.timestamp), 'MMM d, h:mm a')}
              </span>
              <span>{activity.description}</span>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}
```

```tsx
'use client';

import { useState } from 'react';
import { cn } from '@/lib/utils';

interface UsageMetricsChartProps {
  data?: {
    daily: Array<{
      date: string;
      logins: number;
      casesCreated: number;
      casesResolved: number;
      campaignResponses: number;
    }>;
    summary: {
      totalLogins: number;
      uniqueUsers: number;
      avgSessionDuration: number;
      peakDay: string;
    };
  };
}

export function UsageMetricsChart({ data }: UsageMetricsChartProps) {
  const [selectedMetric, setSelectedMetric] = useState<'logins' | 'cases' | 'campaigns'>('logins');

  if (!data) return <div className="h-64 flex items-center justify-center text-gray-500">No data available</div>;

  const maxValue = Math.max(
    ...data.daily.map((d) => {
      if (selectedMetric === 'logins') return d.logins;
      if (selectedMetric === 'cases') return d.casesCreated + d.casesResolved;
      return d.campaignResponses;
    })
  );

  return (
    <div className="space-y-4">
      {/* Metric selector */}
      <div className="flex items-center gap-2">
        {[
          { key: 'logins', label: 'Logins' },
          { key: 'cases', label: 'Cases' },
          { key: 'campaigns', label: 'Campaigns' },
        ].map((metric) => (
          <button
            key={metric.key}
            onClick={() => setSelectedMetric(metric.key as any)}
            className={cn(
              'px-3 py-1 text-sm rounded-full',
              selectedMetric === metric.key
                ? 'bg-blue-100 text-blue-700'
                : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
            )}
          >
            {metric.label}
          </button>
        ))}
      </div>

      {/* Summary stats */}
      <div className="grid grid-cols-4 gap-4">
        <div className="p-3 bg-gray-50 rounded-lg">
          <div className="text-2xl font-bold">{data.summary.totalLogins}</div>
          <div className="text-xs text-gray-500">Total Logins</div>
        </div>
        <div className="p-3 bg-gray-50 rounded-lg">
          <div className="text-2xl font-bold">{data.summary.uniqueUsers}</div>
          <div className="text-xs text-gray-500">Unique Users</div>
        </div>
        <div className="p-3 bg-gray-50 rounded-lg">
          <div className="text-2xl font-bold">{Math.round(data.summary.avgSessionDuration)}m</div>
          <div className="text-xs text-gray-500">Avg Session</div>
        </div>
        <div className="p-3 bg-gray-50 rounded-lg">
          <div className="text-2xl font-bold">{data.summary.peakDay}</div>
          <div className="text-xs text-gray-500">Peak Day</div>
        </div>
      </div>

      {/* Bar chart */}
      <div className="h-48 flex items-end gap-1">
        {data.daily.map((day, i) => {
          const value = selectedMetric === 'logins'
            ? day.logins
            : selectedMetric === 'cases'
            ? day.casesCreated + day.casesResolved
            : day.campaignResponses;
          const height = maxValue > 0 ? (value / maxValue) * 100 : 0;

          return (
            <div
              key={day.date}
              className="flex-1 flex flex-col items-center group"
            >
              <div className="w-full relative">
                <div
                  className="w-full bg-blue-500 rounded-t transition-all group-hover:bg-blue-600"
                  style={{ height: `${height * 1.8}px` }}
                />
                {/* Tooltip */}
                <div className="absolute bottom-full mb-2 left-1/2 -translate-x-1/2 hidden group-hover:block z-10">
                  <div className="bg-gray-900 text-white text-xs px-2 py-1 rounded whitespace-nowrap">
                    {day.date}: {value}
                  </div>
                </div>
              </div>
              {i % 5 === 0 && (
                <div className="text-xs text-gray-400 mt-1">
                  {new Date(day.date).getDate()}
                </div>
              )}
            </div>
          );
        })}
      </div>
    </div>
  );
}
```

```tsx
import { CheckCircle, XCircle } from 'lucide-react';
import { cn } from '@/lib/utils';

interface FeatureAdoptionGridProps {
  features?: Array<{
    key: string;
    name: string;
    category: string;
    adopted: boolean;
    firstUsed?: string;
    lastUsed?: string;
  }>;
}

const categoryOrder = ['Core', 'Advanced', 'Integration', 'Reporting'];

export function FeatureAdoptionGrid({ features }: FeatureAdoptionGridProps) {
  if (!features) return null;

  const groupedFeatures = features.reduce((acc, feature) => {
    if (!acc[feature.category]) acc[feature.category] = [];
    acc[feature.category].push(feature);
    return acc;
  }, {} as Record<string, typeof features>);

  const adoptedCount = features.filter((f) => f.adopted).length;
  const adoptionRate = Math.round((adoptedCount / features.length) * 100);

  return (
    <div className="space-y-6">
      {/* Summary */}
      <div className="flex items-center gap-4">
        <div className="text-2xl font-bold">{adoptionRate}%</div>
        <div className="text-gray-500">
          {adoptedCount} of {features.length} features adopted
        </div>
      </div>

      {/* Feature grid by category */}
      {categoryOrder.map((category) => {
        const categoryFeatures = groupedFeatures[category];
        if (!categoryFeatures) return null;

        return (
          <div key={category}>
            <h3 className="font-medium text-sm text-gray-500 mb-2">{category}</h3>
            <div className="grid grid-cols-3 gap-2">
              {categoryFeatures.map((feature) => (
                <div
                  key={feature.key}
                  className={cn(
                    'p-3 border rounded-lg flex items-center gap-2',
                    feature.adopted ? 'bg-green-50 border-green-200' : 'bg-gray-50'
                  )}
                >
                  {feature.adopted ? (
                    <CheckCircle className="h-4 w-4 text-green-500 flex-shrink-0" />
                  ) : (
                    <XCircle className="h-4 w-4 text-gray-300 flex-shrink-0" />
                  )}
                  <span className={cn(
                    'text-sm',
                    !feature.adopted && 'text-gray-400'
                  )}>
                    {feature.name}
                  </span>
                </div>
              ))}
            </div>
          </div>
        );
```

```tsx
import { cn } from '@/lib/utils';

interface BenchmarkCompareProps {
  title: string;
  metric: string;
  clientValue: number | null;
  benchmarkData?: {
    p25: number;
    median: number;
    p75: number;
    peerCount: number;
  };
  invertComparison?: boolean; // For metrics where lower is better
  unit?: string;
}

export function BenchmarkCompare({
  title,
  metric,
  clientValue,
  benchmarkData,
  invertComparison = false,
  unit = '%',
}: BenchmarkCompareProps) {
  if (!benchmarkData) return null;

  const { p25, median, p75, peerCount } = benchmarkData;
  const range = p75 - p25;
  const min = Math.max(0, p25 - range * 0.5);
  const max = p75 + range * 0.5;

  const getPosition = (value: number) => {
    return ((value - min) / (max - min)) * 100;
  };

  const getPercentile = (value: number) => {
    if (invertComparison) {
      if (value <= p25) return 75 + (p25 - value) / p25 * 25;
      if (value <= median) return 50 + (median - value) / (median - p25) * 25;
      if (value <= p75) return 25 + (p75 - value) / (p75 - median) * 25;
      return Math.max(0, 25 - (value - p75) / p75 * 25);
    } else {
      if (value >= p75) return 75 + (value - p75) / p75 * 25;
      if (value >= median) return 50 + (value - median) / (p75 - median) * 25;
      if (value >= p25) return 25 + (value - p25) / (median - p25) * 25;
      return Math.max(0, (value / p25) * 25);
    }
  };

  const clientPercentile = clientValue !== null ? Math.round(getPercentile(clientValue)) : null;
  const clientPosition = clientValue !== null ? getPosition(clientValue) : null;

  const getPercentileColor = (percentile: number) => {
    if (percentile >= 75) return 'text-green-600';
    if (percentile >= 50) return 'text-blue-600';
    if (percentile >= 25) return 'text-yellow-600';
    return 'text-red-600';
  };

  return (
    <div className="bg-white border rounded-lg p-6">
      <h3 className="font-semibold mb-4">{title}</h3>

      {/* Client value and percentile */}
      {clientValue !== null && clientPercentile !== null && (
        <div className="mb-6">
          <div className="flex items-baseline gap-2">
            <span className="text-3xl font-bold">{clientValue}{unit}</span>
            <span className={cn('text-lg font-medium', getPercentileColor(clientPercentile))}>
              {clientPercentile}th percentile
            </span>
          </div>
        </div>
      )}

      {/* Visual benchmark scale */}
      <div className="relative pt-8 pb-4">
        {/* Scale bar */}
        <div className="h-4 bg-gray-200 rounded-full relative">
          {/* Quartile markers */}
          <div
            className="absolute top-0 bottom-0 bg-gray-300 rounded-l-full"
            style={{ left: `${getPosition(p25)}%`, width: `${getPosition(median) - getPosition(p25)}%` }}
          />
          <div
            className="absolute top-0 bottom-0 bg-blue-200"
            style={{ left: `${getPosition(median)}%`, width: `${getPosition(p75) - getPosition(median)}%` }}
          />

          {/* P25 marker */}
          <div
            className="absolute top-0 -translate-x-1/2"
            style={{ left: `${getPosition(p25)}%` }}
          >
            <div className="w-0.5 h-6 bg-gray-400" />
            <div className="text-xs text-gray-500 mt-1 -ml-4 w-8 text-center">
              {p25}{unit}
            </div>
            <div className="text-xs text-gray-400 -ml-4 w-8 text-center">25th</div>
          </div>

          {/* Median marker */}
          <div
            className="absolute top-0 -translate-x-1/2"
            style={{ left: `${getPosition(median)}%` }}
          >
            <div className="w-0.5 h-6 bg-blue-500" />
            <div className="text-xs text-blue-600 font-medium mt-1 -ml-4 w-8 text-center">
              {median}{unit}
            </div>
            <div className="text-xs text-gray-400 -ml-4 w-8 text-center">median</div>
          </div>

          {/* P75 marker */}
          <div
            className="absolute top-0 -translate-x-1/2"
            style={{ left: `${getPosition(p75)}%` }}
          >
            <div className="w-0.5 h-6 bg-gray-400" />
            <div className="text-xs text-gray-500 mt-1 -ml-4 w-8 text-center">
              {p75}{unit}
            </div>
            <div className="text-xs text-gray-400 -ml-4 w-8 text-center">75th</div>
          </div>

          {/* Client position marker */}
          {clientPosition !== null && (
            <div
              className="absolute -top-6 -translate-x-1/2"
              style={{ left: `${Math.min(100, Math.max(0, clientPosition))}%` }}
            >
              <div className="w-4 h-4 bg-purple-600 rounded-full border-2 border-white shadow-lg" />
              <div className="w-0.5 h-8 bg-purple-600 mx-auto" />
            </div>
          )}
        </div>
      </div>

      {/* Peer count */}
      <div className="text-xs text-gray-400 text-center">
        Compared to {peerCount} organizations
      </div>
    </div>
  );
}
```
